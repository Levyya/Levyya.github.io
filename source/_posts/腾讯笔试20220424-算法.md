##### 下注

$$
(n + x)(1-p)=n\\
n为上一次下注金额，x为跟注金额\\
x = \frac{np}{1-p} \quad or \quad n_{i+1} = \frac{n_i}{1-p}
$$

```c++
#include <bits/stdc++.h>
using namespace std;
int main() {
	long long n, m;
	cin >> n >> m;
	long long ans = n;
	for (int i = 0; i < m; i++) {
		double p;
		cin >> p;
		double jia = (double)ans * p / (1.0 - p);
		if (jia - (long long)jia < 1e-6) {
			jia -= 1;
		}
		ans = ans + jia;
	}
	return 0;
}
```

##### 全连接层（张量计算）

A为c*c矩阵，f为c\*1向量，求Af的最大值对应索引（从0开始）

```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e4 + 7;
double f[N], a[N][N], ans[N];
int main() {
	int c;
	cin >> c;
	for (int i = 0; i < c; i++) {
		cin >> f[i];
	}
	for (int i = 0; i < c; i++) {
		for (int j = 0; j < c; j++) {
			cin >> a[i][j];
		}
	}
	int max_id = 0;
	double max_val = 0;
	for (int i = 0; i < c; i++) {
		for (int j = 0; j < c; j++) {
			ans[i] += a[i][j] * f[j];
		}
		if (ans[i] > max_val) {
			max_val = ans[i];
			max_id = i;
		}
	}
	cout << max_id;
	return 0;
}
```

##### 一元三次方根

```c++
#include <bits/stdc++.h>
using namespace std;
int main() {
	int q;
	long double a, b, c;
	cin >> q;
	while (q--) {
		cin >> a >> b >> c;
		long double l = 1e-8, r = 2e9;
		int f = 0;
		while (r - l > 1e-8) {
			long double mid = (l + r) / 2;
			long double res = pow(mid, 3) + a * pow(mid, 2) + b * mid - c;
			if (abs(res) < 1e-8) {
				printf("%.8llf\n", mid);
				f = 1;
				break;
			} else if (res < 1e-8) {
				l = mid;
			} else {
				r = mid;
			}
		}
		if (f == 0) {
			printf("%.8llf\n", l);
		}
	}
	return 0;
}
```

##### 区间染色

```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 2e5 + 7;
long long a[N], b[N];
int main() {
	int n, m, q;
	cin >> n >> m;
	long long ans1 = 0, ans2 = 0, ans3 = 0;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	for (int i = 0; i < m; i++) {
		cin >> b[i];
	}
	sort(a, a + n);
	sort(b, b + m);
	cin >> q;
	int l, r;
	while (q--) {
		cin >> l >> r;
		long long *ll = lower_bound(a, a + n, l);
		long long *rr = lower_bound(a, a + n, r);
		long long suma = rr - ll;
		if (*rr == r) {
			suma++;
		}
		ll = lower_bound(b, b + m, l);
		rr = lower_bound(b, b + m, r);
		long long sumb = rr - ll;
		if (*rr == r) {
			sumb++;
		}
		if (suma == sumb) {
			ans2++;
		} else if (suma > sumb) {
			ans1++;
		} else {
			ans3++;
		}
	}
	printf("%d %d %d\n", ans1, ans2, ans3);
	return 0;
}
```

##### 公共数链

```python
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None

class Solution:
    def solve(self, root, b):
        def getParent(node, x, y):
            if not node:
                return None
            if node.val == x or node.val == y:
                return node
            left = getParent(node.left, x, y)
            right = getParent(node.right, x, y)
            if left and right:
                return node
            elif left:
                return left
            elif right:
                return right

        def dfs(node, target, res):
            nonlocal path
            if path:
                return
            if not node:
                return
            res.append(node.val)
            if node.val == target:
                path = res[:]
                return
            dfs(node.left, target, res)
            dfs(node.right, target, res)
            res.pop()

        def getSame(list1, list2):
            res = []
            list2 = set(list2)
            for node in list1:
                if node in list2:
                    res.append(ListNode(node))
            if not res:
                return res
            flag = False
            for i in range(len(res)):
                j = len(res) - 1 - i
                if res[i].val == res[j].val:
                    continue
                elif res[i].val < res[j].val:
                    break
                else:
                    flag = True
                    break
            if flag:
                res = res[::-1]
            for i in range(len(res) - 1):
                res[i].next = res[i+1]
            return res[0]


        def printAns(ans):
            for ele in ans:
                while ele:
                    print(ele.val, end=' ')
                    ele = ele.next
                print()

        ans = [None] * len(b)
        for i, (x, y, u, v) in enumerate(b):
            p1 = getParent(root, x, y)
            path = []
            dfs(p1, x, [])
            res1 = path[:]
            path = []
            dfs(p1, y, [])
            res2 = path[:]
            res00 = res1[::-1] + res2[1:]

            p2 = getParent(root, u, v)
            path = []
            dfs(p2, u, [])
            res3 = path[:]
            path = []
            dfs(p2, v, [])
            res4 = path[:]
            res11 = res3[::-1] + res4[1:]
            ans[i] = getSame(res00, res11)
        printAns(ans)
        return ans
```
